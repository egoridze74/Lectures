Написать скрипт на shell для вывода в столбец внешних аргументов при его вызове. Если при вызове скрипта внешние аргументы отсутствуют, то вывести идентификатор текущего процесса.

```
if [ $# -eq 0 ]
then echo идентификатор текущего процесса равен $$
else echo внешние аргументы:
	while [ $1 ]
	do
		echo $1
		shift
	done
fi

# Это был код файла. Теперь идет командная строка:
# Сохранили файл под названием lab

chmod u +x lab
./lab # выведет идентификатор процесса
./lab one two 123 # выведет внешние аргументы в столбик
```
