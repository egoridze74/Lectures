## Задание 1
Необходимо написать программу для определения типа файла по запросу. Предусмотреть обработку сигнала прерывания от клавы. При поступлении 3 сигналов

``` shell
i = 0 #счетчик
flag = 0 #фиксатор ввода

handle()
{
	i = `expr $i + 1`
	if [ $i -eq 3 ]
	then file * | grep "C source" | cut -d : -f 1
	#или
	then file * | awk '{ print $1 }'
	fi
	
	if [ $flag -eq 0 ]
	then echo "Введите имя файла"
	fi
}

trap "handle" SIGINT
echo "Введите имя файла"
read name
flag = 1

if test -e $name
then 
	b = `ls -lda $name | cut -c 1-1`
	case $b in
	'_') echo "файл $name - обычный";;
	'd') echo "файл $name - директория";;
	'p') echo "файл $name - канал";;
	... #c, b, s, l
	'*') default;;
	esac
else
	echo "файла с таким именем не существует"
fi

sleep 1
sleep 1
sleep 1
sleep 1
```
